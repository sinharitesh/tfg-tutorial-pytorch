def do_training(model, num_epochs = 100):
    losses = []
    for i in range(num_epochs):
        y_hat = model(X)
        loss = mse(y_hat, y)
        losses.append(loss)
        if i % 20 == 0: print(i, loss)
        with torch.no_grad():
            loss.backward()
            for param in model.parameters():
                param.sub_(lr * param.grad)
                param.grad.zero_()
    return(model, losses)
